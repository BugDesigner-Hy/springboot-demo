# springboot-demo
spring boot 实践学习案例

目前已有的模块有
```xml
    <modules>
        <!--简单的springboot初探-->
        <module>springboot-helloworld</module>
        <!-- swagger2的增强版-->
        <module>springboot-knife4j</module>
        <!-- mybatis-plus-->
        <module>springboot-mybatis-plus</module>
        <!-- druid 数据库池化管理-->
        <module>springboot-druid</module>
        <!--spring security安全框架-->
        <module>springboot-security</module>
        <!--OAuth2第三方登录-->
        <module>springboot-oauth</module>
    </modules>
```
